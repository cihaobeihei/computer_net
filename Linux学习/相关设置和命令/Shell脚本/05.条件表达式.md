## 2.case语句
一个实例：<br>
```shell
#!/bin/bash
##############################################################
# File Name: color.sh
# Version: V1.0
# Author: noble
# Created Time : 2018-10-10 15:40:58
# Description:
##############################################################
function colorString()
{
    local string=$1
    local color=$2
    if [ -z "$string" ]
    then
        echo "1Usege:$0 (string) (color)"
        exit 1
    elif [ -z "$color" ]
    then
        echo "2Usege:$0 (string) (color)"
        exit 1
    fi
    case $color in
    black)
        echo -e "\033[30m${string}\033[0m"
        ;;
    red)
        echo -e "\033[31m${string}\033[0m"
        ;;
    green)
        echo -e "\033[32m${string}\033[0m"
        ;;
    yellow)
        echo -e "\033[33m${string}\033[0m"
        ;;
    blue)
        echo -e "\033[34m${string}\033[0m"
        ;;
    purple)
        echo -e "\033[35m${string}\033[0m"
        ;;
    *)
        echo "Unsupport"
        exit 1
    esac
}
```
