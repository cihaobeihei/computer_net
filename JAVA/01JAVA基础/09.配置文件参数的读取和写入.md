# 配置文件的读取和写入
程序在运行中有一些参数需要从配置文件中读取。<br>
我们一般使用Map容器(键值对)来维护这些参数，在java中有一个现成类是从配置文件中读取写入参数的是properties类。<br>

## 使用实例
```java
package com.noble;

import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.InputStreamReader;
import java.io.OutputStreamWriter;
import java.io.Reader;
import java.io.Writer;
import java.util.Properties;

import org.junit.jupiter.api.Test;

class JunitTest {

	@Test
	public void demo01() throws Exception{//写入参数
		Properties props = new Properties();

		props.setProperty("k01", "v01");
		props.setProperty("k02", "v02");
		props.setProperty("k03", "v03");

		Writer writer = new OutputStreamWriter(new FileOutputStream("F:\\properties"));
		props.store(writer, "describe");//执行写操作
		writer.close();
	}

	@Test
	public  void demo02() throws Exception {//读取参数
		Reader reader = new InputStreamReader(new FileInputStream("F:\\properties"));
		Properties props = new Properties();

		props.load(reader);//执行读操作

		for(String name : props.stringPropertyNames()) {
			String value = props.getProperty(name);
			System.out.println(name + ":" + value);
		}
	}

	public static void main(String[] args) throws Exception {
		//demo02();
		JunitTest j = new JunitTest();
		j.demo02();
	}
}

```
