## 1.创建用户并随机设置密码
```shell
#!/bin/bash
##############################################################
# File Name: test3.sh
# Version: V1.0
# Author: noble
# Created Time : 2018-10-13 19:54:09
# Description:
##############################################################
for i in {01..10}
do
    pass=`mkpasswd -l 10 -d 0 -c 0 -C 0 -s 0`
    #useradd noble$i
    echo $pass|passwd --stdin noble$i
    echo "noble${i}:${pass}" >> /tmp/passwd.list
done
```

```shell
#!/bin/bash
##############################################################
# File Name: test3.sh
# Version: V1.0
# Author: noble
# Created Time : 2018-10-13 19:54:09
# Description:
##############################################################
for n in `seq -w 11 15`
do
    pass=`openssl rand -base64 9`
    useradd noble$n
    echo "noble${n}:${pass}" >> /tmp/passwd.log
done
chpasswd < /tmp/passwd.log
```

## 在for循环中做到并行处理

```shell
#!/bin/bash
##############################################################
# File Name: test.sh
# Version: V1.0
# Author: noble
# Created Time : 2018-10-13 10:20:24
# Description:
##############################################################
for n in {1..254}
do
    {
        if `ping -c 1 10.0.0.$n &>/dev/null`
        then
            echo "10.0.0.$n up..."
        else
            echo "10.0.0.$n down..."
        fi
    }&
done
```
通过使用大括号``{}&``实现并行。<br>
